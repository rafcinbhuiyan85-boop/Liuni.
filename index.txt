// MUSIC AUTO PLAY
window.addEventListener("load", function() {

    document.body.addEventListener("click", () => {
      audio.play();
    }, { once: true });
  });
});

// ROSE FUNCTION
function spawnRose(x, y) {
  const rose = document.createElement("div");
  rose.className = "rose";
  rose.innerHTML = "🌹";
  rose.style.left = x + "px";
  rose.style.top = y + "px";
  document.body.appendChild(rose);
  setTimeout(() => rose.remove(), 4000);
}

// POPUP TEXT
function showPopup() {
  const popup = document.createElement("div");
  popup.className = "popup";
  popup.innerText = "Ei naw, tmr jonno ekti golap🌹";
  document.body.appendChild(popup);
  setTimeout(() => popup.remove(), 3000);
}

// CLICK EVENT
document.getElementById("mainImg").addEventListener("click", function(e) {
  spawnRose(e.clientX, e.clientY);
  showPopup();
});

// AUTO ROSE
setInterval(() => {
  spawnRose(Math.random() * window.innerWidth, window.innerHeight);
}, 600);
